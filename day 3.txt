Nested lists

students=[]
if __name__ == '__main__':
    for _ in range(int(input())):
        name = input()
        score = float(input())
        students.append([name, score])
second = sorted(set(score for name, score in students))[1]
for name, score in sorted(students):
    if score == second:
        print(name)
